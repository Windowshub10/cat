1️⃣ Daily / Weekly / Monthly / Quarterly profit of each store

Daily

SELECT 
[Measures].[Profit] ON COLUMNS,
([Dim Store].[Store Name].Members *
 [Dim Date].[Date].Members) ON ROWS
FROM [Sales DW]

Weekly

SELECT 
[Measures].[Profit] ON COLUMNS,
([Dim Store].[Store Name].Members *
 [Dim Date].[Week].Members) ON ROWS
FROM [Sales DW]

Monthly

SELECT 
[Measures].[Profit] ON COLUMNS,
([Dim Store].[Store Name].Members *
 [Dim Date].[Month].Members) ON ROWS
FROM [Sales DW]

Quarterly

SELECT 
[Measures].[Profit] ON COLUMNS,
([Dim Store].[Store Name].Members *
 [Dim Date].[Quarter].Members) ON ROWS
FROM [Sales DW]


---

2️⃣ Comparison of sales and profit on various time periods

SELECT
{
[Measures].[Sales],
[Measures].[Profit]
} ON COLUMNS,
[Dim Date].[Month].Members ON ROWS
FROM [Sales DW]


---

3️⃣ Comparison of sales in various time bands of the day

SELECT
[Measures].[Sales] ON COLUMNS,
[Dim Time].[Time Band].Members ON ROWS
FROM [Sales DW]


---

4️⃣ Which product has more demand on which location

SELECT
[Measures].[Quantity] ON COLUMNS,
([Dim Product].[Product Name].Members *
 [Dim Store].[Store Name].Members) ON ROWS
FROM [Sales DW]


---

5️⃣ Trend of sales by time period of the day over week / month / year

SELECT
[Measures].[Sales] ON COLUMNS,
([Dim Time].[Time Band].Members *
 [Dim Date].[Month].Members) ON ROWS
FROM [Sales DW]


---

6️⃣ On what day sales is higher

SELECT
[Measures].[Sales] ON COLUMNS,
[Dim Date].[Day Name].Members ON ROWS
FROM [Sales DW]


---

7️⃣ On every Sunday of this month, sales and profit

SELECT
{
[Measures].[Sales],
[Measures].[Profit]
} ON COLUMNS,
[Dim Date].[Date].Members ON ROWS
FROM [Sales DW]
WHERE ([Dim Date].[Day Name].[Sunday])


---

8️⃣ Trend of sales on weekday and weekend

SELECT
[Measures].[Sales] ON COLUMNS,
[Dim Date].[Day Type].Members ON ROWS
FROM [Sales DW]
