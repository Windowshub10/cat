### Sales Total Cost from cube

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS
FROM [Sales DW]


### Same basic Sales query

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS
FROM [Sales DW]


###  Sales for Year 2013

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS
FROM [Sales DW]
WHERE ([Dim Date].[Year].&[2013])


### Sales grouped by Year dimension

SELECT
[Measures].[Sales Total Cost] ON COLUMNS,
NON EMPTY ([Dim Date].[Year].[Year]) ON ROWS
FROM [Sales DW]

### Display sales by year

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS,
[Dim Date].[Year].[Year].MEMBERS ON ROWS
FROM [Sales DW]

### Display sales and quantity together

SELECT 
{
[Measures].[Sales Total Cost],
[Measures].[Quantity]
} ON COLUMNS
FROM [Sales DW]


### Sales by store and product (CrossJoin)

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS,
CROSSJOIN(
[Dim Stores].[Store ID].[Store ID].MEMBERS,
[Dim Product].[Product Key].[Product Key].MEMBERS
) ON ROWS
FROM [Sales DW]


### Top 5 sales days

SELECT 
[Measures].[Sales Total Cost] ON COLUMNS,
TOPCOUNT(
[Dim Date].[Date].[Date].MEMBERS,
5,
[Measures].[Sales Total Cost]
) ON ROWS
FROM [Sales DW]
