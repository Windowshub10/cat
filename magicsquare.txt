def create_magic_square(n):
    if n % 2 == 0:
        print("Sorry, this only works for odd numbers.")
        return None
    square = [[0] * n for _ in range(n)]
    row = 0
    col = n // 2
    for number in range(1, n * n + 1):
        square[row][col] = number
        next_row = (row - 1) % n
        next_col = (col + 1) % n

        if square[next_row][next_col] != 0:
            row = (row + 1) % n
        else:
            row = next_row
            col = next_col
    return square

size = 3
magic_square = create_magic_square(size)

print(f"Here is a {size}x{size} Magic Square:")

if magic_square: 
    for row_list in magic_square:
        for number in row_list:
            print(" ",number, end=" ")
        print()
