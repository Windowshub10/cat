// jQuery Insert Content, jQuery Remove Elements and Attribute. 


<!DOCTYPE html> 
<html> 
<head> 
<title>Simple jQuery Demo</title> 
<script 
src="https://ajax.googleapis.com/ajax/libs/jquery/3.7.1/jquery.min.js"></script> 
<style> 
body { font-family: sans-serif; text-align: center; padding-top: 20px; } 
#box { border: 2px solid #000; padding: 20px; margin: 20px; min-height: 100px; 
} 
button { margin: 4px; padding: 10px; } 
.highlight { background-color: yellow; border-color: orange; } 
p.new { color: green; font-weight: bold; } 
span { color: blue; } 
</style> 
</head> 
<body> 
<h2>My Box</h2> 
<div id="box"> 
<p>This is a paragraph inside the box. <a href="#">A link</a></p> 
</div> 
<hr> 
<h3>Click the buttons to change the box above!</h3> 
<div> 
<button id="btn-append">Append</button> 
<button id="btn-prepend">Prepend</button> 
<button id="btn-after">Add After</button> 
<button id="btn-before">Add Before</button> 
</div> 
<div> 
<button id="btn-remove-p">Remove Paragraphs</button> 
<button id="btn-empty">Empty Box</button> 
</div> 
<div> 
<button id="btn-change-attr">Change Link Attribute</button> 
<button id="btn-toggle-class">Toggle Highlight Class</button> 
</div> 
<script> 
$(document).ready(function(){ 
// --- INSERT CONTENT --- 
$('#btn-append').click(function(){ 
$('#box').append("<span>(Appended Text)</span> "); 
}); 
 
$('#btn-prepend').click(function(){ 
$('#box').prepend("<span>(Prepended Text)</span> "); 
}); 
$('#btn-after').click(function(){ 
$('#box').after('<p class="new">-- Added AFTER the box --</p>'); 
}); 
$('#btn-before').click(function(){ 
$('#box').before('<p class="new">-- Added BEFORE the box --</p>'); 
}); 
// --- REMOVE ELEMENTS --- 
$('#btn-remove-p').click(function(){ 
// This removes any paragraph inside the box 
$('#box p').remove(); 
}); 
$('#btn-empty').click(function(){ 
// This removes everything inside the box 
$('#box').empty(); 
}); 
// --- ATTRIBUTES & CLASSES --- 
$('#btn-change-attr').click(function(){ 
// Change the link's href attribute and its text 
$('#box a').attr('href', 'https://jquery.com').text('jQuery Website'); 
}); 
$('#btn-toggle-class').click(function(){ 
// Add or remove the 'highlight' class from the box 
$('#box').toggleClass('highlight'); 
}); 
}); 
</script> 
</body> 
</html>
