# Use group by(), %>%,mutate(), rename(),arrange(), filter(), select()

# Install and Load dplyr
install.packages("dplyr")
library(dplyr)

# Create Sample Dataset
data <- data.frame(
  Name = c("Parth", "Siya", "Vansh", "Dev", "Diya","Alex","Vansh"),
  Age = c(20, 22, 21, 23, 20,21,21),
  Marks = c(75, 88, 64, 91, 75,90,64),
  Gender = c("M", "F", "M", "M", "F","M","M")
)

# View Data
data

# FILTER (Select rows)
filter(data, Age > 21)
filter(data, Marks > 70 & Gender == "F")

# SELECT (Select columns)
select(data, Name, Marks)          # Select specific columns
select(data, Name:Marks)           # Select range
select(data, -Gender)              # Exclude column
select(data, starts_with("M"))     # Select column starting with M
select(data, ends_with("e"))       # Select column ending with e

# SUMMARY (Automatic summary)
summary(data) #Shows:- Minimum, 1st Quartile, Median, Mean, 3rd Quartile, Maximum

# SUMMARIZE (Custom summary)
summarize(data,
          total_marks = sum(Marks),
          avg_age = mean(Age))


# ARRANGE (Sorting)
arrange(data, Marks)               # Ascending
arrange(data, desc(Name))         # Descending

# RENAME (Change column name)
rename(data, Student_Name = Name)

# MUTATE (Add new column, keep old)
mutate(data, Total = Age + Marks)

# TRANSMUTE (New column only)
transmute(data, Total = Age + Marks)

# GROUP BY + SUMMARIZE
data %>%
  group_by(Gender) %>%
  summarize(avg_marks = mean(Marks),
            total_students = n())

# PIPE OPERATOR (%>%)
data %>%
  filter(Marks > 70) %>%
  arrange(desc(Marks)) %>%
  select(Name, Marks)

# DISTINCT (Remove duplicates)
data
distinct(data, Name)   # Unique names
distinct(data)         # Unique full rows
