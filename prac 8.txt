## Five number summary in exploratory data analysis


data <- c(25, 34, 28, 48, 52, 37, 29, 35, 54, 43, 64, 39, 58, 45)

fivenum(data) #Shows:- Minimum, 1st Quartile, Median, 3rd Quartile, Maximum

summary(data) #Shows:- Minimum, 1st Quartile, Median, Mean, 3rd Quartile, Maximum

boxplot(data,col="orange")
