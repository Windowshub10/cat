## Multiple histogram and multiple boxplots, Multiple scatter plots and colouring the graph


# Multiple Boxplats
boxplot(mpg ~ cyl, data = mtcars, 
        main = "MPG by Number of Cylinders",
        col = c("red", "green", "blue"),
        xlab = "Cylinders", ylab = "Miles Per Gallon")
mtcars


# Multiple Histograms
par(mfrow = c(1, 3)) 
hist(mtcars$mpg, col = "skyblue", main = "Histogram of MPG")
hist(mtcars$hp, col = "orange", main = "Histogram of Horsepower")
hist(mtcars$disp, col = "yellow", main = "Histogram of Displacement")
par(mfrow = c(1, 1))


# Multiple Scatter plots
par(mfrow = c(1, 3))
plot(mtcars$wt, mtcars$mpg, 
     main = "Weight vs MPG", 
     xlab = "Weight", ylab = "MPG", 
     pch = 19, col = "skyblue")
plot(mtcars$hp, mtcars$mpg, 
     main = "Horsepower vs MPG", 
     xlab = "Horsepower", ylab = "MPG", 
     pch = 19, col = "orange")
plot(mtcars$cyl, mtcars$mpg, 
     main = "Cylinder vs MPG", 
     xlab = "Horsepower", ylab = "MPG", 
     pch = 19, col = "yellow")
par(mfrow = c(1, 1))


# Coloring The graph with data
plot(mtcars$wt, mtcars$mpg, 
     col = as.factor(mtcars$cyl), 
     pch = 19,                    
     main = "Weight vs MPG (Colored by Cylinders)",
     xlab = "Weight", 
     ylab = "MPG")
legend("topright", legend = levels(as.factor(mtcars$cyl)), 
       fill = 1:3, title = "Cylinders")
